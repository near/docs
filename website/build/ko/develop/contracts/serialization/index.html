  vector: Vector<u8>
}

#[near_bindgen]
impl Contract {
  #[init]
  pub fn init(string: String, first_u8: u8) -> Self {
    let mut vector: Vector<u8> = Vector::new("prefix".as_bytes());
    vector.push(&first_u8);

    Self { string, vector }
  }

    self.string = string;
    self.vector.push(&number);
  }
